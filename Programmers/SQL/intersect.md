```mysql
SELECT DISTINCT A.CART_ID ID
FROM
   (SELECT CART_ID
   FROM CART_PRODUCTS
    WHERE NAME = 'MILK') A
JOIN
   (SELECT CART_ID
   FROM CART_PRODUCTS
    WHERE NAME = 'Yogurt') B
ON A.CART_ID = B.CART_ID
ORDER BY A.CART_ID

# Using IN
SELECT CART_ID
FROM CART_PRODUCTS
WHERE NAME = "Milk" AND CART_ID in
    (SELECT CART_ID 
    FROM CART_PRODUCTS 
    WHERE NAME = "Yogurt")
ORDER BY CART_ID

# FROM 2
SELECT A.CART_ID FROM 
 (SELECT CART_ID FROM CART_PRODUCTS WHERE NAME = 'Yogurt') as A, 
 (SELECT CART_ID FROM CART_PRODUCTS WHERE NAME = 'Milk') as B
WHERE A.CART_ID = B.CART_ID

SELECT DISTINCT C.CART_ID 
FROM CART_PRODUCTS C, CART_PRODUCTS T
    WHERE C.CART_ID = T.CART_ID 
        AND (C.NAME ='Milk' AND T.NAME ='Yogurt')
            ORDER BY C.CART_ID
```
